括号匹配

这次用顺序栈实现的括号匹配，不多说，直接上代码：

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define  SIZE 100
#define  N 10

typedef struct ST{
    char *top;
    char *base;
    int  size;
}stack;

void Init_st(stack &l)
{
    l.base = (char *)malloc(sizeof(char) * SIZE);
    l.top = l.base;
    l.size = SIZE;
}

int Empty_st(stack &l)
{
    if(l.base == l.top)
        return 1;
    else
        return 0;
}

void Push_st(stack &l,char e)
{    
    *(l.top++) = e;
}

void Pop_st(stack &l)
{
    if(l.top == l.base)
         return;
    else
         l.top--;
}    

char Getop_st(stack &l)
{
    if(l.top == l.base)
          exit(0);
    else 
          return *(--l.top);
}

void Checksymbol(stack &l,char *a)
{
    int flag = 1;

    for(int i = 0; i < strlen(a); i++)
        if(a[i] == '(' || a[i] == '[' || a[i] == '{')
                Push_st(l,a[i]);
        else if(a[i] == ')')
        {
             if(Empty_st(l) || Getop_st(l) != '(')   flag = 0;                    
             else           Pop_st(l);
        }                   
        else if(a[i] == ']')
        {
             if(Empty_st(l) || Getop_st(l) != '[')   flag = 0;              
             else           Pop_st(l);
        }      
        else if(a[i] == '}')       
        {
             if(Empty_st(l) || Getop_st(l) != '{')   flag = 0;              
             else           Pop_st(l);
        }

    if(!Empty_st(l))  flag = 0;

    if(flag)
         printf("Yes\n");
    else
         printf("No\n");
}

int main()
{
    char ch[100];

    while(~scanf("%s",ch))
    {
        stack s;
        Init_st(s);
        Checksymbol(s,ch);
    }

    return 0;
}
